۴. Pydantic AI — Agents و Tools

۴.۱ analysis_agent

from pydantic import BaseModel
from pydantic_ai import Agent

class MeetingAnalysis(BaseModel):
    title: str
    summary: str
    key_points: list[str]
    decisions: list[str]
    tasks: list[MeetingTask]

analysis_agent = Agent(
    'google:gemini-2.5-flash',
    output_type=MeetingAnalysis,
    system_prompt='تحلیل transcript جلسه فارسی. فقط بر اساس متن.',
)

۴.۲ rag_agent

rag_agent = Agent(
    'google:gemini-2.5-flash',
    output_type=RagAnswer,
    deps_type=MeetingDeps,
)

@rag_agent.tool
async def search_meeting_chunks(ctx, query: str, top_k: int = 5) -> list[str]:
    return ctx.deps.vector_store.search(ctx.deps.meeting_id, query, top_k)
Jira create از API route مستقیم است (نه agent) — رفتار predictable برای دمو و خطای کمتر.
